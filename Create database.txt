Создаю бд:
create database DataAnomic;



Подключаюсь к ней:
use DataAnomic;



Создаю таблицу UserData:
CREATE TABLE UserData( 
id INT NOT NULL AUTO_INCREMENT,
role VARCHAR(30), 
email VARCHAR(50),
password VARCHAR(50), 
firstname VARCHAR(30),
lastname VARCHAR(30), 
office VARCHAR(30),
birthdate date, 
state boolean,
active datetime,
lastactive datetime, 
PRIMARY KEY(id));
